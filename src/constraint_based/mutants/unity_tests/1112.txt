select student_id, min(course_id) as course_id, grade from Enrollments where (student_id, grade) in (select student_id, max(grade) from Enrollments group by student_id) group by student_id, grade order by student_id
select student_id, min(course_id) as course_id, grade from Enrollments where 1 = 1 group by student_id, grade order by student_id 
select student_id, min(course_id) as course_id, grade from Enrollments where (student_id, grade) in (select student_id, max(distinct grade) from Enrollments group by student_id) group by student_id, grade order by student_id
select student_id, min(distinct course_id) as course_id, grade from Enrollments where (student_id, grade) in (select student_id, max(grade) from Enrollments group by student_id) group by student_id, grade order by student_id
